# Differences between JS and Python on Arithmetic Operators.


### `+` Addition operator

```js
//JavaScript
let a = 10;
let b = 5;
let x = a + b;
console.log(x);
15
```

```py
#Python
>>> a = 8
>>> b = 2
>>> c = a + b
>>> c
10
```
---

### `-` Subtraction operator

```js
//JavaScript
let a = 20;
let b = 8;
let x = a - b;
console.log(x);
12
```

```py
#Python
>>> a = 10
>>> b = 3
>>> c = a - b
>>> c
7
```
---

### `*` Multiplication operator

```js
//JavaScript
let a = 2;
let b = 8;
let x = a * b;
console.log(x);
16
```

```py
#Python
>>> a = 5
>>> b = 8
>>> c = a * b
>>> c
40
```
---

### `**` Exponential operator

```js
//JavaScript
let x = 5;
// 5 will be divided 2 times.
let y = 2;
let z = x ** y;
console.log(z)
25
x ** y
25
```

```py
#Python
>>> a = 4
# 4 will be divided 3 times.
>>> b = 3
>>> c = a ** b
>>> c
64
```
---

```js

```

```py
#Python
>>> a = 20
>>> b = 5
```
